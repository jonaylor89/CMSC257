
IP = argv[1]

myServer = socket(PF_INET, SOCK_STREAM, 0)

bind(myServer, ServerIP, PORT)

listen(myServer, 5)

while true 
  client = accept(myServer)

  if fork() == 0 

    FILENAME = read(myServer, client, 50)  

    if not exists(FILENAME) 
      write(myServer, client, "/cmsc257")
    endif

    fd = open(FILENAME, 'r')
    while still data

      dataChunk = read(fd, 50)
      write(myServer, client, dataChunk) 

    endwhile

    write(myServer, client, "/cmsc257")

    close(client)

    exit(0)
  else
    continue 

  endif

endwhile
