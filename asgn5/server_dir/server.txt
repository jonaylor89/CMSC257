
IP = argv[1]

myServer = socket(PF_INET, SOCK_STREAM)

bind(myServer, ServerIP, PORT)

while true 
  client = accept(myServer)

  if fork() == 0 
    while true
      FILENAME = recv(client)  
      open(FILENAME, 'r')
      while still data
        dataChunk = read(FILENAME, length)
        send(myServer, client, dataChunk) 
      endwhile
    endwhile
    close(client)
  else
    continue 
  endif
endwhile
